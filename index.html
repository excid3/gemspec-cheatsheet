GemSpec CheatSheet

Gem::Specification.new do |s|
  s.platform          = Gem::Platform::RUBY
  s.name              = 'example'
  s.rubyforge_project = 'example'
  s.version           = '0.4.1'
  s.summary           = 'An example gem.'
  s.description       = 'An example gem with many gemspec options.'
  s.requirements      = 'A powerful graphics card.'

  s.required_ruby_version     = '>= 1.9.3'
  s.required_rubygems_version = '>= 1.8.11'

  s.author            = 'John Jones'
  s.authors           = ['John Jones', 'Mary Smith']
  s.email             = 'david@loudthinking.com'
  s.homepage          = 'http://www.rubyonrails.org'
  s.license           = 'MIT'
  s.licenses          = ['MIT', 'GPL-2']

  s.has_rdoc         = true
  s.rdoc_options     << '--line-numbers'
  s.extra_rdoc_files = ['README', 'doc/user-guide.txt']

  s.bindir             = 'bin'
  s.executables        = ['bin/foo', 'bin/bar']
  s.default_executable = 'bin/bar'

  s.require_path  = '.'
  s.require_paths << 'ext'
  s.files         = FileList['lib/**/*.rb', 'test/**/*'].to_a
  s.test_files    = Dir.glob('test/tc_*.rb')

  s.add_dependency             'bundler',         '~> 1.1'
  s.add_development_dependency 'rspec'

  s.extensions << 'ext/rmagic/extconf.rb'
end
